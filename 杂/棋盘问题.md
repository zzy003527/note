# 棋盘问题
```c
include <stdio.h>

int main()
{
	int n = 0, m = 0;
	int n1, m1;
	int x = 0, x1 = 0, x2 = 0;
	scanf_s("%d%d", &n, &m);
	n1 = n;
	m1 = m;
	while (n1 != 0 && m1 != 0)
	{
		x1 += n1 * m1;
		n1--;
		m1--;
	}                                                //正方形个数
	x = ((n + 1) * n * (m + 1) * m) / 4;             //矩形总数
		x2 = x - x1;                                 //长方形个数
		printf("%d %d", x1, x2);
	return 0;
}
```
