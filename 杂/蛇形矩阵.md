## 蛇形矩阵

```c
# include <stdio.h>
int main()
{
	int n = 0;
	int a[10][10] = { 0 };
	int i = 1;
	int k = 0, j = 0;
	scanf_s("%d", &n);
	int dx[] = { -1,0,1,0 };
	int dy[] = { 0,1,0,-1 };
	int x = 0, y = 0, d = 1;
	int p = 0, q = 0;
	for (i = 1; i <= n*n; i++)
	{
		a[x][y] = i;
		p = x + dx[d];
		q = y + dy[d];
		if (p < 0 || q < 0 || p >= n || q >= n || a[p][q])    //当超出矩阵或者碰到已有元素时改变方向
		{
			d = (d + 1) % 4;
			p = x + dx[d];
			q = y + dy[d];
		}
		x = p;
		y = q;
	}
	for (k = 0; k < n; k++)                            //输出矩阵
	{
		for (j = 0; j < n; j++)
		{
			printf("%3d", a[k][j]);
		}
		printf("\n");                                   //此处不写无法输出矩阵
	}
	return 0;
}
```
