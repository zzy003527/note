## 黑白棋子的移动

```c
# include<stdio.h>
# include<string.h>
void print1();
void print2();
char chess[1200];
int i;
int n;
int main()
{
	scanf_s("%d", &n);
	for (i = 1; i <= n; i++)
	{
		chess[i] = 'o';
	}
	for (i = n + 1 ; i <= 2 * n; i++)
	{
		chess[i] = '*';
	}
	chess[2 * n + 1] = '-';
	chess[2 * n + 2] = '-';     //定义第一行棋子顺序
	int m = n;
	while (m >= 4)                 //输出除了最后四行外的其他棋子
	{
		print1();
		chess[2 * m + 1] = 'o';
		chess[2 * m + 2] = '*';
		chess[m] = chess[m + 1] = '-';
		print1();
		m--;
		chess[2 * m + 1] = chess[2 * m + 2] = '-';
		chess[m + 1] = chess[m + 2] = '*';
	}
	print2();
	return 0;
}
void print1()                                //输出每一行棋子
{
	for (i = 1; i <= 2 * n + 2; i++)
	{
		printf("%c", chess[i]);
	}
	printf("\n");                 
}
void print2()                     //输出最后四行棋子
{
	printf("%s", "ooo*o**--*");
	for (i = 4; i < n; i++)
	{
		printf("%s", "o*");
	}
	printf("\n");
	printf("%s", "o--*o**oo*");
	for (i = 4; i < n; i++)
	{
		printf("%s", "o*");
	}
	printf("\n");
	printf("%s", "o*o*o*--o*");
	for (i = 4; i < n; i++)
	{
		printf("%s", "o*");
	}
	printf("\n");
	printf("%s", "--o*o*o*o*");
	for (i = 4; i < n; i++)
	{
		printf("%s", "o*");
	}
}
```
